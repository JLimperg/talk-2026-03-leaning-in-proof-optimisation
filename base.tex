\begin{frame}
  \frametitle{Context}

  \begin{block}{Objective}
    Fix obvious deficiencies in Lean proofs:
    \begin{itemize}
      \item Unused \lean|have| tactic
      \item Tactic never called
      \item Manual proof can be replaced by \lean|grind|/\lean|linarith|/\lean|aesop|
      \item Non-terminal \lean|aesop| or non-squeezed \lean|simp|
      \item \dots
    \end{itemize}
  \end{block}

  \pause

  \begin{block}{Motivation}
    \begin{itemize}
      \item Improve AI-generated proofs for training and human consumption
      \item (Eventually) improve sloppy human proofs and auto-apply linter suggestions
    \end{itemize}
  \end{block}
\end{frame}

\begin{frame}[fragile]
  \frametitle{An LLM-Generated Proof}

  \begin{leancode}[escapeinside=$$]
    theorem cos_range_value (a : ℝ) (x : ℝ)
        (h_cos : cos x = (2 * a - 3) / (4 - a))
        (h_quad : -1 < cos x ∧ cos x < 0) :
        -1 < a ∧ a < 3 / 2 := by
      have h_denom_ne_zero : 4 - a ≠ 0 := by
        by_contra h
        have h₁ : 4 - a = 0 := by
          linarith
        rw [h₁] at h_cos
        norm_num at h_cos
        have h₂ : cos x = 0 := by
          linarith
        linarith [h_quad.2]
      have h_denom_pos : 4 - a > 0 := by

      $\emph{... 104 lines later ...}$

      exact ⟨h_neg_one_lt_a, h_a_lt_three_half⟩
  \end{leancode}

  Produced by DeepSeek-Prover-V2.
\end{frame}

\begin{frame}
  \frametitle{Symbolic Proof Optimisation}

  \begin{itemize}
    \item Similar to an optimising compiler.
    \item Applies various \emph{optimisation passes} that simplify the proof.
    \item Implemented in an internal (hopefully eventually open source) Lean library.
  \end{itemize}

  \pause

  Mathlib's tactic analysis framework\footnote{Mathlib/Tactic/TacticAnalysis.lean} is conceptually similar, but only generates optimisation suggestions without applying them.
\end{frame}

\begin{frame}
  \frametitle{Why Not LLM?}

  \begin{itemize}
    \item Use specially trained LLM-based optimisers\footfullcite{ProofOptimizer}, or just ask Claude.
    \item Tradeoff: LLMs are more general, less development effort, more costly (time and money).
    \item Combined approach may be optimal: symbolic for obvious defects, LLMs for more interesting optimisations.
    \item Can integrate LLMs (or SLMs) as optimisation passes in a symbolic framework.
  \end{itemize}
\end{frame}

  \frametitle{Redundancy 1: Unused Facts}

  \begin{leancode}[escapeinside=$$]
    have h₆ : a > 3 / 2 := by

    $\emph{... h₆ never used ...}$
  \end{leancode}
\end{frame}

\begin{frame}[fragile]
  \frametitle{Redundancy 2: Restating Known Facts}

  \begin{leancode}
    have h_denom_ne_zero : 4 - a ≠ 0 := by
      ...

    have h₂ : 4 - a ≠ 0 := h_denom_ne_zero
  \end{leancode}
\end{frame}

\begin{frame}[fragile]
  \frametitle{Redundancy 3: Restating the Goal}

  \begin{leancode}[escapeinside=$$]
    have h₅₂ : $\emphbox{(2 * a - 3) / (4 - a) ≥ 0}$ := by
      have h₅₃ : 4 - a < 0 := h₃
      have h₅₄ : $\emphbox{(2 * a - 3) / (4 - a) ≥ 0}$ := by
        have h₅₅ : 2 * a - 3 ≤ 0 := h₅₁
        have h₅₆ : 4 - a < 0 := h₃
        have h₅₇ : $\emphbox{(2 * a - 3) / (4 - a) ≥ 0}$ := by
          ...
        exact h₅₇
      exact h₅₄
  \end{leancode}
\end{frame}

\begin{frame}[fragile]
  \frametitle{Redundancy 4: Unused Tactics}
  \begin{leancode}
    field_simp [h₅₁, sub_ne_zero.mpr h₅₁] <;>
      ring_nf <;>
      field_simp [h₅₁, sub_ne_zero.mpr h₅₁] <;>
      nlinarith
  \end{leancode}
\end{frame}

\begin{frame}
  \begin{block}{Results}
    \begin{itemize}
      \item Reduces our example proof from \emph{115} lines to \emph{60}.
      \item Reduces code size of one of our SFT datasets (also produced by DeepSeek-Prover-V2\footfullcite{DSPv2}) by \emph{60\%}.
    \end{itemize}
  \end{block}

  \begin{block}{Caveats}
    \begin{itemize}
      \item Optimisation passes are generally \emph{incomplete}.
      \item Optimisation may \emph{hurt} when training an LLM to \emph{find} Lean proofs in one shot.
    \end{itemize}
  \end{block}
\end{frame}

\begin{frame}
  \frametitle{Architecture}

  We distinguish three kinds of optimisation passes:
  \begin{itemize}
    \item \emph{Syntactic}: purely syntactic, guaranteed semantics-preserving.
    \item \emph{Semantic}: need elaboration information, e.g.\ goals before/after tactics.
      \begin{itemize}
        \item \emph{Unchecked}: guaranteed semantics-preserving.
        \item \emph{Checked}: generally not semantics-preserving; need to check individual rewrites.
      \end{itemize}
  \end{itemize}
\end{frame}

\begin{frame}[fragile]
  \frametitle{Syntactic Passes}

  \begin{block}{Examples}
    \begin{leancode}
      simp; skip     →   simp
      solve | fail   →   fail
      simp at h h    →   simp at h
    \end{leancode}

    \pause

    \begin{leancode}
      theorem foo (n : Nat) : (m : Nat) → P n m := by
        intro m
        ...

      →

      theorem foo (n m : Nat) : P n m := by
        ...
    \end{leancode}
  \end{block}
\end{frame}

\begin{frame}
  \frametitle{Syntactic Passes}

  \begin{itemize}
    \item Purpose: clean up after more interesting optimisations, e.g.\ after replacing an unused tactic with \lean|skip|.
    \item Very cheap (compared to semantic passes), so we apply them before/after every semantic pass.
    \item We make them even cheaper:
      \begin{itemize}
        \item Combine all optimisations into a single post-traversal with a fixpoint loop at each syntax node.
        \item Index individual optimisations by their head node kind.
      \end{itemize}
  \end{itemize}
\end{frame}

\begin{frame}[fragile]
  \frametitle{Custom Syntactic Sugar for Syntactic Passes}

  We define a custom elaborator for syntactic optimisations.

  \begin{leancode}
    def solve := ls_analysis_fn%
      | `(tactic| solve $[| $tss]*) => do
        let tss' := tss.filter fun ts =>
          ! (isSkipTacticSeq ts || isFailTacticSeq ts)
        if tss'.size == tss.size then
          return none
        if tss'.isEmpty then
          `(tactic| fail)
        else
          `(tactic| solve $[| $tss']*)
  \end{leancode}
\end{frame}

\begin{frame}
  \frametitle{Semantic Passes}

  Rely on the \emph{info trees} produced during elaboration.
  These contain information about expression types, goals before/after each tactic, identifier shadowing, etc.

  \pause

  E.g. to remove \lean|have h : A := ...|, we must know that \lean|h| was not used during the elaboration of the following tactics.
\end{frame}

\begin{frame}
  \frametitle{Info Tree Constraints}

  \begin{block}{Info tree nodes are identified by syntax source position}
    Hence every rewrite invalidates the info trees.
  \end{block}

  \pause

  \begin{block}{Syntax lookup is linear}
    Probably not a problem in practice, but we prematurely optimise by lazily building indices for the information we need.
  \end{block}

  \pause

  \begin{block}{Info tree information is sometimes incomplete}
    E.g.\ the connection between names in the syntax and fvars is sometimes lost.
  \end{block}

  \pause

  Info trees are intended for use by the language server, so it's expected that they do not perfectly support our use case.
\end{frame}

\begin{frame}
  \frametitle{The Performance Challenge}

  Naively, we can \emph{re-elaborate the command after each rewrite}, both to check the rewrite and to update the info trees.

  \begin{block}{Problem}
    Prohibitively \emph{expensive}: elaborating a proof can take \emph{seconds}.
  \end{block}

  \begin{block}{Partial solution: batching}
    Perform \emph{multiple rewrites} and only then elaborate the changed command.

    \begin{itemize}
      \item Only works for non-checked passes.
      \item Only works if the rewrites are guaranteed not to interfere with each other.
    \end{itemize}
  \end{block}

  \pause

  \begin{block}{Partial solution (future work): compositional optimisation}
    We ought to make use of the typical decomposition of Lean proofs into \emph{focused subproofs}.
    However, there are subtleties, e.g.\ a focused subproof can have non-local effects if its initial goal contains a metavariable.
  \end{block}
\end{frame}

\begin{frame}
  \frametitle{The Extensibility Challenge}

  \begin{block}{Problem}
    Almost any invariant we rely on can be violated by custom tactics.
  \end{block}

  \begin{block}{Solution (future work)}
    Define \emph{extension hooks} for all passes that instruct the pass to skip certain tactics or process them differently.
    (Many linters have similar hooks.)
  \end{block}

  \pause

  \begin{block}{Solution (for now)}
    Ignore the problem and accept a certain failure rate (< 5\% on our SFT dataset).
  \end{block}
\end{frame}

\begin{frame}
  \frametitle{A Final Caveat}

  The pretty-printer somewhat regularly produces syntax that does not elaborate correctly, even with the \lean|pp.analyze| option.
  This is a known (and very hard) problem that will hopefully be solved at some point.
\end{frame}
